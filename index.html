Here is a solution to a problem from Codewars: 'You are going to be given a word. Your job is to return the middle character of the word. If the word's length is odd, return the middle character. If the word's length is even, return the middle 2 characters.' 

            
              function getMiddle(s)
              {
                if (s.length % 2 === 0) {
                  const middlePoint = s.length / 2 - 1;
                  return s.charAt(middlePoint) + s.charAt(middlePoint + 1)
                } else {
                  const middlePoint = s.length / 2 - 1;
                  return s.charAt(middlePoint + 1)
                }
              }
